#President Impeached
country_event = {
	id = usa.1
	title = usa.1.t
	desc = usa.1.d
	
	is_triggered_only =  yes
	fire_only_once = no
	
	option = {
		name = usa.1.a
		if = {
			limit = {
			    has_country_leader = { name = "Donald Trump" ruling_only = yes } 
			}
			create_country_leader = {
	            name = "Mike Pence"
	            picture = "Mike_Pence.dds"
	            ideology = national_conservative
            }
			else = {
			    retire_country_leader = yes
			}
		}
	}
}
