add_namespace = syria

#End of the Syrian Civil War - Assad
news_event = {
	id = syria.1
	title = syria.1.t
	desc = syria.1.d
	picture = GFX_news_event_syrian_war_over
	
	is_triggered_only = yes
	fire_only_once = no
	major = yes
	
	option = {
		name = syria.1.o1
		every_state = {
		    limit = {
		        OR = {
					is_core_of = FSA
					is_core_of = HTS
					is_claimed_by = FSA
					is_claimed_by = HTS
				}
			}
			remove_core_of = FSA
			remove_core_of = HTS
		    remove_claim_by = FSA
			remove_claim_by = HTS
		}
	}
}
